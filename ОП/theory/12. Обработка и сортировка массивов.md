3. swap двух элементов массива

```c
void swap(int* a, int* b){
    int temp = *b;
    *b = *a;
    *a = temp;
}
```

4. Сортировка полным перебором

```c
void sort(int* arr, int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = i + 1; j < n; j++)
            if (arr[i] > arr[j])
                swap(arr + i, arr + j);
}
```

5. Сортировка пузырьком

```c
void sort(int* arr, int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - i - 1; j++)
            if (arr[j] > arr[j + 1])
                swap(arr + j, arr + j + 1);
}
```

6. Сортировка массива методом поиска минимального элемента массива

```c
 void sort(double* arr, int n) {
    for (int i = 0; i < n - 1; i++) {
        int indexMin = i;
        for (int j = i + 1; j < n; j++)
            if (arr[indexMin] > arr[j])
                indexMin = j;
        swap(arr + i, arr + indexMin);
    }
}
```
