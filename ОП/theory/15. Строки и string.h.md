# 15. Строки. Строковой и символьный литералы. Представление в памяти. Примеры различных способов инициализации строк. Основные функции библиотеки \<string.h\> и их реализация.

В языке Си строк НЕТ. Под строками понимается массив символов.<br>Строковый литерал " " – это последовательность символов в двойных кавычках (строковая константа) или массив символов, заканчивающийся терминальным нулем \0.<br>Символьный литерал ' ' – один символ в одинарных кавычках.

Статическое выделение:<br>char string\[41\]; - создан массив из 41 символа типа char<br>char string\[\] = "Горные вершины спят во тьме ночной.";

char\* str = (char\*)malloc(n \* sizeof(char));

Основные функции string.h

1. strlen - определение длины строки

```c
unsigned strLen(char* s) {
    unsigned count = 0;
    while (s[count])
        count++;
    return count;
}
```

1. strcpy - копирование строки

```c
char* strCpy(char* s1, const char* s2){
    for (int i = 0; (s1[i] = s2[i]); i++);
    return s1;
}
```

1. strcat - объединение двух строк

```c
char* strCat(char* s1, const char* s2){
    char* p = s1;
    for (; *p; p++);
    for (int i = 0; (p[i] = s2[i]); i++);
    return s1;
}
```

1. strcmp - сравнение строк

```c
int strCmp(const char* s1, const char* s2) {
    int i = 0;
    for (; s1[i] && s1[i] == s2[i]; i++);
    return s1[i] - s2[i];
}
```

5.strchr - поиск первого вхождения символа

```c
 char* strChr(const char* s, int c){
    const char* p = s;
    for(; *p && *p != c; p++);
    return (*p == c)? (char*)p : NULL;
}
```

1. strstr - поиск первого вхождения подстроки

```c
char* strStr(char* s, const char* subs){
    int isFound = 0;
    char* p = s;
    for (; !isFound && *p; ){
        int i = 0;
        for (; subs[i] && p[i] == subs[i]; i++);
        if (!subs[i])
            isFound = 1;
        else
            p++;
    }
    return isFound ? p : 0;
}
```

1. strpbrk - поиск первого символа из последовательности

```c
char* strPbrk(char* s, const char* sym){
    int isFound = 0;
    char* p = s;
    for (; !isFound && *p;){
        unsigned i = 0;
        for (; sym[i] && *p != sym[i]; i++);
        if (sym[i])
            isFound = 1;
        else
            p++;
    }
    return isFound ? p : 0;
}
```

1. strspn - длина начальной подстроки из допустимых символов

```c
size_t my_strspn(const char* s, const char* sym) {
    int isFound = 0;
    size_t len = 0;
    for (; !isFound && s[len]; ) {
        const char* p;
        for (p = sym; *p && s[len] != *p; p++);
        if (!*p)
            isFound = 1;
        else
            len++;
    }
    return len;
}
```

1. strdel - удаление подстроки в строке

```c
char* strDel(char* s, int n){
    if ((int)strLen(s) <= n)
        *s = 0;
    else
        strCpy(s, s + n);
    return s;
}
```

1. strins - вставка подстроки в строку

```c
void strIns(char* str, char* sub){
    int lenStr = strLen(str);
    int lenSub = strLen(sub);
    for (int i = 0; i < lenSub; i++)
        str[lenStr + i] = sub[i];
    str[lenStr + lenSub] = '\0';
}
```

void strSubMy(char* subs, const char* s, unsigned len){
